--Using comparison operators

--QUERY emp_no for those employees making less 
--or equal to 78000 from salaries table

select emp_no, salary
from salaries
where salary <= 78000;

--QUERY emp_no for those employees making more
--or equal to 81000 from salaries table

select emp_no, salary
from salaries
where salary >= 81000;


--BETWEEN: condition will return the records where the expression is
--within the rage of value and value2
--QUERY from_date, emp_no, salary from those employees
--who worked between 1987-01-01 until 1994-01-01

select emp_no, from_date, salary
from salaries
where to_date between '1987-01-01' and '1994-01-01';
